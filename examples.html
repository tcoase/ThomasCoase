Example 1: Solving a First-Order ODE with Euler's Method

This example solves the differential equation \(y' = -2y\) with initial condition \(y(0) = 1\) using Euler's method. The exact solution is \(y(t) = e^{-2t}\).

% Solve y' = -2y, y(0) = 1 using Euler's method
% Define parameters
f = @(t, y) -2*y;  % The differential equation
t0 = 0;            % Initial time
y0 = 1;            % Initial condition
h = 0.1;           % Step size
t_final = 1;       % Final time

% Calculate number of steps
n_steps = ceil((t_final - t0) / h);

% Initialize arrays
t = zeros(n_steps + 1, 1);
y_euler = zeros(n_steps + 1, 1);
y_exact = zeros(n_steps + 1, 1);

% Set initial values
t(1) = t0;
y_euler(1) = y0;
y_exact(1) = y0;

% Euler's method iteration
for i = 1:n_steps
    % Update time
    t(i+1) = t(i) + h;
    
    % Euler step: y_{n+1} = y_n + h*f(t_n, y_n)
    y_euler(i+1) = y_euler(i) + h * f(t(i), y_euler(i));
    
    % Compute exact solution for comparison
    y_exact(i+1) = exp(-2*t(i+1));
end

% Calculate error
error = abs(y_euler - y_exact);
max_error = max(error);

% Display results
fprintf('Initial value problem: y'' = -2y, y(0) = 1\n');
fprintf('Exact solution: y = exp(-2t)\n\n');
fprintf('Step size: %.1f\n', h);

for i = 1:n_steps+1
    if i == 1 || i == 2 || i == 3 || i == 4 || i == 5 || i == 6 || i == n_steps+1
        fprintf('t = %.1f, y_euler = %.4f, y_exact = %.4f, error = %.4e\n', ...
                t(i), y_euler(i), y_exact(i), error(i));
        if i == 6
            fprintf('...\n');
        end
    end
end

fprintf('\nGlobal error (maximum): %.4e\n', max_error);

% Plot results
figure;
plot(t, y_euler, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Euler');
hold on;
plot(t, y_exact, 'r-', 'LineWidth', 1.5, 'DisplayName', 'Exact');
xlabel('t');
ylabel('y');
title('Solution of y'' = -2y, y(0) = 1');
legend('show');
grid on;

Example 2: Comparing Different Step Sizes

This example demonstrates how the accuracy of Euler's method improves with smaller step sizes by solving the same differential equation with different step sizes.

% Compare Euler's method with different step sizes
% Solve y' = -2y, y(0) = 1

% Define parameters
f = @(t, y) -2*y;  % The differential equation
t0 = 0;            % Initial time
y0 = 1;            % Initial condition
t_final = 1;       % Final time

% Define different step sizes
step_sizes = [0.2, 0.1, 0.05, 0.025];
colors = {'b', 'g', 'm', 'c'};

% Initialize figure
figure;
hold on;

% Loop through different step sizes
for j = 1:length(step_sizes)
    h = step_sizes(j);
    
    % Calculate number of steps
    n_steps = ceil((t_final - t0) / h);
    
    % Initialize arrays
    t = zeros(n_steps + 1, 1);
    y_euler = zeros(n_steps + 1, 1);
    
    % Set initial values
    t(1) = t0;
    y_euler(1) = y0;
    
    % Euler's method iteration
    for i = 1:n_steps
        % Update time
        t(i+1) = t(i) + h;
        
        % Euler step
        y_euler(i+1) = y_euler(i) + h * f(t(i), y_euler(i));
    end
    
    % Plot results for this step size
    plot(t, y_euler, [colors{j}, '-o'], 'LineWidth', 1.5, ...
         'DisplayName', ['h = ', num2str(h)]);
    
    % Calculate error at final time
    exact_final = exp(-2*t_final);
    error_final = abs(y_euler(end) - exact_final);
    fprintf('Step size h = %.3f, Final error = %.6e\n', h, error_final);
end

% Plot exact solution
t_exact = linspace(t0, t_final, 100);
y_exact = exp(-2*t_exact);
plot(t_exact, y_exact, 'r-', 'LineWidth', 1.5, 'DisplayName', 'Exact');

% Add labels and legend
xlabel('t');
ylabel('y');
title('Euler''s Method with Different Step Sizes');
legend('show');
grid on;

% Create error convergence plot
figure;
errors = zeros(length(step_sizes), 1);
for j = 1:length(step_sizes)
    h = step_sizes(j);
    n_steps = ceil((t_final - t0) / h);
    
    % Initialize arrays
    t = zeros(n_steps + 1, 1);
    y_euler = zeros(n_steps + 1, 1);
    
    % Set initial values
    t(1) = t0;
    y_euler(1) = y0;
    
    % Euler's method iteration
    for i = 1:n_steps
        t(i+1) = t(i) + h;
        y_euler(i+1) = y_euler(i) + h * f(t(i), y_euler(i));
    end
    
    % Calculate error at final time
    exact_final = exp(-2*t_final);
    errors(j) = abs(y_euler(end) - exact_final);
end

% Plot error vs step size on log-log scale
loglog(step_sizes, errors, 'bo-', 'LineWidth', 1.5);
hold on;

% Plot reference line with slope 1 (first-order convergence)
h_ref = logspace(log10(min(step_sizes)), log10(max(step_sizes)), 100);
err_ref = h_ref * (errors(end) / step_sizes(end));
loglog(h_ref, err_ref, 'r--', 'LineWidth', 1.5, 'DisplayName', 'First-order (slope = 1)');

xlabel('Step Size (h)');
ylabel('Error at t = 1');
title('Error Convergence of Euler''s Method');
legend('Error', 'First-order reference', 'Location', 'northwest');
grid on;

Example 1: Solving a First-Order ODE with Heun's Method

This example solves the differential equation \(y' = y - t^2 + 1\) with initial condition \(y(0) = 0.5\) using Heun's method. The exact solution is \(y(t) = (t + 1)^2 - 0.5e^t\).

% Solve y' = y - t^2 + 1, y(0) = 0.5 using Heun's method
% Define parameters
f = @(t, y) y - t^2 + 1;  % The differential equation
t0 = 0;                   % Initial time
y0 = 0.5;                 % Initial condition
h = 0.1;                  % Step size
t_final = 1;              % Final time

% Calculate number of steps
n_steps = ceil((t_final - t0) / h);

% Initialize arrays
t = zeros(n_steps + 1, 1);
y_heun = zeros(n_steps + 1, 1);
y_exact = zeros(n_steps + 1, 1);

% Set initial values
t(1) = t0;
y_heun(1) = y0;
y_exact(1) = y0;

% Heun's method iteration
for i = 1:n_steps
    % Update time
    t(i+1) = t(i) + h;
    
    % Predictor step (Euler)
    y_pred = y_heun(i) + h * f(t(i), y_heun(i));
    
    % Corrector step (Trapezoidal)
    y_heun(i+1) = y_heun(i) + h/2 * (f(t(i), y_heun(i)) + f(t(i+1), y_pred));
    
    % Compute exact solution for comparison
    y_exact(i+1) = (t(i+1) + 1)^2 - 0.5*exp(t(i+1));
end

% Calculate error
error = abs(y_heun - y_exact);
max_error = max(error);

% Display results
fprintf('Initial value problem: y'' = y - t^2 + 1, y(0) = 0.5\n');
fprintf('Exact solution: y = (t + 1)^2 - 0.5*exp(t)\n\n');
fprintf('Step size: %.1f\n', h);

for i = 1:n_steps+1
    if i == 1 || i == 2 || i == 3 || i == 4 || i == 5 || i == 6 || i == n_steps+1
        fprintf('t = %.1f, y_heun = %.4f, y_exact = %.4f, error = %.4e\n', ...
                t(i), y_heun(i), y_exact(i), error(i));
        if i == 6
            fprintf('...\n');
        end
    end
end

fprintf('\nGlobal error (maximum): %.4e\n', max_error);

% Plot results
figure;
plot(t, y_heun, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Heun');
hold on;
plot(t, y_exact, 'r-', 'LineWidth', 1.5, 'DisplayName', 'Exact');
xlabel('t');
ylabel('y');
title('Solution of y'' = y - t^2 + 1, y(0) = 0.5');
legend('show');
grid on;

Example 2: Comparing Euler's and Heun's Methods

This example compares the accuracy of Euler's method and Heun's method for solving the same differential equation.

% Compare Euler's and Heun's methods
% Solve y' = y - t^2 + 1, y(0) = 0.5

% Define parameters
f = @(t, y) y - t^2 + 1;  % The differential equation
t0 = 0;                   % Initial time
y0 = 0.5;                 % Initial condition
h = 0.1;                  % Step size
t_final = 1;              % Final time

% Calculate number of steps
n_steps = ceil((t_final - t0) / h);

% Initialize arrays
t = zeros(n_steps + 1, 1);
y_euler = zeros(n_steps + 1, 1);
y_heun = zeros(n_steps + 1, 1);
y_exact = zeros(n_steps + 1, 1);

% Set initial values
t(1) = t0;
y_euler(1) = y0;
y_heun(1) = y0;
y_exact(1) = y0;

% Iteration for both methods
for i = 1:n_steps
    % Update time
    t(i+1) = t(i) + h;
    
    % Euler's method
    y_euler(i+1) = y_euler(i) + h * f(t(i), y_euler(i));
    
    % Heun's method
    y_pred = y_heun(i) + h * f(t(i), y_heun(i));
    y_heun(i+1) = y_heun(i) + h/2 * (f(t(i), y_heun(i)) + f(t(i+1), y_pred));
    
    % Compute exact solution
    y_exact(i+1) = (t(i+1) + 1)^2 - 0.5*exp(t(i+1));
end

% Calculate errors
error_euler = abs(y_euler - y_exact);
error_heun = abs(y_heun - y_exact);
max_error_euler = max(error_euler);
max_error_heun = max(error_heun);

% Display results
fprintf('Comparison of Euler''s and Heun''s methods\n');
fprintf('Step size: %.1f\n\n', h);
fprintf('Method\t\tMax Error\tOrder\n');
fprintf('Euler\t\t%.4e\t1\n', max_error_euler);
fprintf('Heun\t\t%.4e\t2\n', max_error_heun);
fprintf('Improvement: %.2f times\n', max_error_euler/max_error_heun);

% Plot results
figure;
plot(t, y_euler, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Euler');
hold on;
plot(t, y_heun, 'g-s', 'LineWidth', 1.5, 'DisplayName', 'Heun');
plot(t, y_exact, 'r-', 'LineWidth', 1.5, 'DisplayName', 'Exact');
xlabel('t');
ylabel('y');
title('Comparison of Euler''s and Heun''s Methods');
legend('show');
grid on;

% Plot errors
figure;
semilogy(t, error_euler, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Euler Error');
hold on;
semilogy(t, error_heun, 'g-s', 'LineWidth', 1.5, 'DisplayName', 'Heun Error');
xlabel('t');
ylabel('Error (log scale)');
title('Error Comparison of Euler''s and Heun''s Methods');
legend('show');
grid on;

% Compare convergence rates
step_sizes = [0.2, 0.1, 0.05, 0.025, 0.0125];
errors_euler = zeros(length(step_sizes), 1);
errors_heun = zeros(length(step_sizes), 1);

for j = 1:length(step_sizes)
    h_j = step_sizes(j);
    n_steps_j = ceil((t_final - t0) / h_j);
    
    % Initialize arrays
    t_j = zeros(n_steps_j + 1, 1);
    y_euler_j = zeros(n_steps_j + 1, 1);
    y_heun_j = zeros(n_steps_j + 1, 1);
    
    % Set initial values
    t_j(1) = t0;
    y_euler_j(1) = y0;
    y_heun_j(1) = y0;
    
    % Iteration for both methods
    for i = 1:n_steps_j
        t_j(i+1) = t_j(i) + h_j;
        
        % Euler's method
        y_euler_j(i+1) = y_euler_j(i) + h_j * f(t_j(i), y_euler_j(i));
        
        % Heun's method
        y_pred_j = y_heun_j(i) + h_j * f(t_j(i), y_heun_j(i));
        y_heun_j(i+1) = y_heun_j(i) + h_j/2 * (f(t_j(i), y_heun_j(i)) + f(t_j(i+1), y_pred_j));
    end
    
    % Calculate errors at final time
    exact_final = (t_final + 1)^2 - 0.5*exp(t_final);
    errors_euler(j) = abs(y_euler_j(end) - exact_final);
    errors_heun(j) = abs(y_heun_j(end) - exact_final);
end

% Plot convergence rates
figure;
loglog(step_sizes, errors_euler, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Euler');
hold on;
loglog(step_sizes, errors_heun, 'g-s', 'LineWidth', 1.5, 'DisplayName', 'Heun');

% Add reference lines
h_ref = logspace(log10(min(step_sizes)), log10(max(step_sizes)), 100);
err_ref1 = h_ref * (errors_euler(end) / step_sizes(end));
err_ref2 = h_ref.^2 * (errors_heun(end) / step_sizes(end)^2);
loglog(h_ref, err_ref1, 'b--', 'LineWidth', 1, 'DisplayName', 'First-order');
loglog(h_ref, err_ref2, 'g--', 'LineWidth', 1, 'DisplayName', 'Second-order');

xlabel('Step Size (h)');
ylabel('Error at t = 1');
title('Convergence Rates of Euler''s and Heun''s Methods');
legend('show');
grid on;

Example 1: Finding Square Root Using Newton-Raphson Method

This example uses the Newton-Raphson method to compute the square root of 2 by finding the root of the equation \(f(x) = x^2 - 2 = 0\).

% Find square root of 2 using Newton-Raphson method
a = 2;  % Find square root of 2

% Define function and its derivative
f = @(x) x^2 - a;
df = @(x) 2*x;

% Initial guess
x0 = 1;

% Apply Newton-Raphson method
function [root, iterations, convergence] = newton_raphson(f, df, x0, tol, max_iter)
    % Initialize variables
    x = x0;
    convergence = zeros(max_iter + 1, 1);
    convergence(1) = x;
    
    % Main iteration loop
    for i = 1:max_iter
        % Compute function value and derivative
        f_val = f(x);
        df_val = df(x);
        
        % Check if derivative is close to zero
        if abs(df_val) < eps
            warning('Derivative close to zero. Method may not converge.');
            df_val = sign(df_val) * eps;  % Avoid division by zero
        end
        
        % Compute next approximation
        x_next = x - f_val / df_val;
        
        % Store current approximation
        convergence(i + 1) = x_next;
        
        % Check for convergence
        if abs(x_next - x) < tol
            root = x_next;
            iterations = i;
            convergence = convergence(1:i + 1);
            return;
        end
        
        % Update current approximation
        x = x_next;
    end
    
    % If reached maximum iterations without converging
    warning('Maximum iterations reached. Method may not have converged.');
    root = x;
    iterations = max_iter;
    convergence = convergence;
end

% Call the function
[root, iterations, convergence] = newton_raphson(f, df, x0, 1e-10, 100);

% Display results
fprintf('Finding square root of %d using Newton-Raphson method\n\n', a);
fprintf('Iteration\tApproximation\t\tError\n');
fprintf('-------------------------------------------------\n');
for i = 1:iterations+1
    fprintf('%d\t\t%.10f\t\t%.2e\n', i-1, convergence(i), abs(convergence(i) - sqrt(a)));
end

fprintf('\nSquare root of %g found by Newton-Raphson: %.10f\n', a, root);
fprintf('Actual square root: %.10f\n', sqrt(a));
fprintf('Error: %.2e\n', abs(root - sqrt(a)));
fprintf('Number of iterations: %d\n', iterations);

% Plot convergence
figure;
semilogy(0:iterations, abs(convergence - sqrt(a)), 'o-', 'LineWidth', 1.5);
xlabel('Iteration');
ylabel('Error');
title('Convergence of Newton-Raphson Method for Square Root');
grid on;

Example 2: Solving a System of Nonlinear Equations

This example uses the Newton-Raphson method to solve a system of nonlinear equations: \(x^2 + y^2 = 4\) and \(e^x - y = 1\).

% Solve a system of nonlinear equations using Newton-Raphson method
% x^2 + y^2 = 4
% e^x - y = 1

% Define the system of equations
F = @(x) [x(1)^2 + x(2)^2 - 4; exp(x(1)) - x(2) - 1];

% Define the Jacobian matrix
J = @(x) [2*x(1), 2*x(2); exp(x(1)), -1];

% Initial guess
x0 = [1; 1];

% Apply Newton-Raphson method for systems
function [x, iterations, convergence] = newton_raphson_system(F, J, x0, tol, max_iter)
    % Initialize variables
    x = x0(:);  % Ensure column vector
    n = length(x);
    convergence = cell(max_iter + 1, 1);
    convergence{1} = x;
    
    % Main iteration loop
    for i = 1:max_iter
        % Compute function value and Jacobian
        F_val = F(x);
        J_val = J(x);
        
        % Check if Jacobian is singular
        if rcond(J_val) < eps
            warning('Jacobian is nearly singular. Method may not converge.');
            % Could use regularization or other techniques here
        end
        
        % Solve linear system J(x) * dx = -F(x)
        dx = J_val \ (-F_val);
        
        % Compute next approximation
        x_next = x + dx;
        
        % Store current approximation
        convergence{i + 1} = x_next;
        
        % Check for convergence
        if norm(dx) < tol
            x = x_next;
            iterations = i;
            convergence = convergence(1:i + 1);
            return;
        end
        
        % Update current approximation
        x = x_next;
    end
    
    % If reached maximum iterations without converging
    warning('Maximum iterations reached. Method may not have converged.');
    iterations = max_iter;
    convergence = convergence;
end

% Call the function
[x, iterations, convergence] = newton_raphson_system(F, J, x0, 1e-10, 100);

% Display results
fprintf('Solution of nonlinear system using Newton-Raphson method:\n');
fprintf('x = %.10f, y = %.10f\n', x(1), x(2));

% Verify the solution
F_val = F(x);
fprintf('\nVerification (should be close to zero):\n');
fprintf('Equation 1: x^2 + y^2 - 4 = %.2e\n', F_val(1));
fprintf('Equation 2: e^x - y - 1 = %.2e\n', F_val(2));
fprintf('Number of iterations: %d\n', iterations);

% Plot the system and solution
figure;
% Create a grid of points
[X, Y] = meshgrid(linspace(-3, 3, 100), linspace(-3, 3, 100));
Z1 = X.^2 + Y.^2 - 4;
Z2 = exp(X) - Y - 1;

% Plot the contours
contour(X, Y, Z1, [0, 0], 'b-', 'LineWidth', 2, 'DisplayName', 'x^2 + y^2 = 4');
hold on;
contour(X, Y, Z2, [0, 0], 'r-', 'LineWidth', 2, 'DisplayName', 'e^x - y = 1');

% Mark the solution
plot(x(1), x(2), 'go', 'MarkerSize', 10, 'LineWidth', 2, 'DisplayName', 'Solution');

% Mark the initial guess
plot(x0(1), x0(2), 'mo', 'MarkerSize', 8, 'LineWidth', 2, 'DisplayName', 'Initial Guess');

% Plot the convergence path
for i = 1:length(convergence)-1
    x_i = convergence{i};
    x_i_plus_1 = convergence{i+1};
    plot([x_i(1), x_i_plus_1(1)], [x_i(2), x_i_plus_1(2)], 'k-o', 'LineWidth', 1);
end

xlabel('x');
ylabel('y');
title('Solution of Nonlinear System');
legend('show');
grid on;
axis equal;

% Plot convergence
figure;
errors = zeros(iterations, 1);
for i = 1:iterations
    errors(i) = norm([convergence{i+1}(1) - x(1); convergence{i+1}(2) - x(2)]);
end

semilogy(1:iterations, errors, 'o-', 'LineWidth', 1.5);
xlabel('Iteration');
ylabel('Error (norm)');
title('Convergence of Newton-Raphson Method for Nonlinear System');
grid on;

% Compare performance: vectorized vs. loop
n = 1000000;
x = rand(1, n);
y = zeros(1, n);

% Using a loop
tic
for i = 1:n
    y(i) = sin(x(i));
end
loop_time = toc;

% Using vectorized operation
tic
y = sin(x);
vectorized_time = toc;
fprintf('Time using vectorized operation: %.6f seconds\n', vectorized_time);
fprintf('Speedup factor: %.2f times faster\n', loop_time/vectorized_time);

Example 1: Creating and Using Functions

This example demonstrates how to create and use functions in MATLAB for solving ODEs.

% Creating and using functions in MATLAB

% Define a function to solve an ODE using Euler's method
function [t, y] = solve_ode(f, t0, t_final, h, y0)
    % SOLVE_ODE Solve an ODE using Euler's method
    %   [T, Y] = SOLVE_ODE(F, T0, T_FINAL, H, Y0) solves the ODE y' = F(t,y)
    %   with initial condition Y0 at time T0 until time T_FINAL using step size H.
    %
    %   Inputs:
    %       F - Function handle for the right-hand side of the ODE
    %       T0 - Initial time
    %       T_FINAL - Final time
    %       H - Step size
    %       Y0 - Initial condition
    %
    %   Outputs:
    %       T - Vector of time points
    %       Y - Vector of solution values
    
    % Calculate number of steps
    n_steps = ceil((t_final - t0) / h);
    
    % Initialize arrays
    t = zeros(n_steps + 1, 1);
    y = zeros(n_steps + 1, 1);
    
    % Set initial values
    t(1) = t0;
    y(1) = y0;
    
    % Euler's method iteration
    for i = 1:n_steps
        % Update time
        t(i+1) = t(i) + h;
        
        % Euler step
        y(i+1) = y(i) + h * f(t(i), y(i));
    end
end

% Use the function to solve y' = -2y, y(0) = 1
f = @(t, y) -2*y;
[t, y] = solve_ode(f, 0, 1, 0.1, 1);

% Display results
disp('Solution of y'' = -2y, y(0) = 1:');
disp([t, y]);

% Compare with exact solution
exact = exp(-2*t);
error = abs(y - exact);
max_error = max(error);
fprintf('Maximum error: %.4e\n', max_error);

% Plot results
figure;
plot(t, y, 'b-o', 'LineWidth', 1.5, 'DisplayName', 'Numerical');
hold on;
plot(t, exact, 'r-', 'LineWidth', 1.5, 'DisplayName', 'Exact');
xlabel('t');
ylabel('y');
title('Solution of y'' = -2y, y(0) = 1');
legend('show');
grid on;

% Function handles for numerical methods

    % Integrate a function using the trapezoidal rule
    h = (b - a) / n;
    x = a:h:b;
    y = arrayfun(f, x);
    result = h * (sum(y) - 0.5*(y(1) + y(end)));
end

% Solution to Challenge: Implement the Newton-Raphson Method
% Find the root of f(x) = x^3 - 2*x - 5

% Define the function and its derivative
f = @(x) x^3 - 2*x - 5;
df = @(x) 3*x^2 - 2;

% Initial guess
x0 = 2;

% Tolerance and maximum iterations
tol = 1e-10;
max_iter = 100;

% Implement the Newton-Raphson method
% 1. Initialize variables
x = x0;
convergence = zeros(max_iter + 1, 1);
convergence(1) = x;

% 2. Iterate until convergence or maximum iterations
for i = 1:max_iter
    % a. Compute function value and derivative
    f_val = f(x);
    df_val = df(x);
    
    % Check if derivative is close to zero
    if abs(df_val) < eps
        warning('Derivative close to zero. Method may not converge.');
        df_val = sign(df_val) * eps;  % Avoid division by zero
    end
    
    % b. Compute next approximation using Newton-Raphson formula
    x_next = x - f_val / df_val;
    
    % Store current approximation
    convergence(i + 1) = x_next;
    
    % c. Check for convergence
    if abs(x_next - x) < tol
        root = x_next;
        iterations = i;
        convergence = convergence(1:i + 1);
        break;
    end
    
    % d. Update current approximation
    x = x_next;
    
    % If reached maximum iterations without converging
    if i == max_iter
        warning('Maximum iterations reached. Method may not have converged.');
        root = x;
        iterations = max_iter;
        convergence = convergence;
    end
end

% Display results
fprintf('Root found: %.10f\n', root);
fprintf('Function value at root: %.2e\n', f(root));
fprintf('Number of iterations: %d\n', iterations);

% Plot convergence
figure;
